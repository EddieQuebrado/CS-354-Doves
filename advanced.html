Let's start with something basic. How about a program which rolls a certain kind of dice a certain number of times, specified by the user's input:

import random #import the random module

print("Roll a set of dice.")
strnumber = input("How many dice? >") #ask to specify how many dice to roll
max = input("How many sides do the dice have? > ") #ask what the maximum number of sides this dice has

printout = str(strnumber)+'d'+str(max) #make a little string to show (in standard D&D notation) what dice you're rolling
print("rolling",printout) #print string we just made

total = 0 #create a variable (with the proper enclosing scope) to keep track of the roll total
number=int(strnumber) #get an int version of the string we got earler

for x in range(number): #loop through as many times as the user specified
y = int(random.randint(1,int(max))) #generate a random number between 1 and what the user specified
print(y) #print it
total = total + y #update the total

print("total:", total) #tell the total to the user
