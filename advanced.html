Let's start with something basic. How about a program which rolls a certain kind of dice a certain number of times, specified by the user's input:

import random #import the random module

print("Roll a set of dice.")
strnumber = input("How many dice? >") #ask to specify how many dice to roll
max = input("How many sides do the dice have? > ") #ask what the maximum number of sides this dice has

printout = str(strnumber)+'d'+str(max) #make a little string to show (in standard D&D notation) what dice you're rolling
print("rolling",printout) #print string we just made

total = 0 #create a variable (with the proper enclosing scope) to keep track of the roll total
number=int(strnumber) #get an int version of the string we got earler

for x in range(number): #loop through as many times as the user specified
y = int(random.randint(1,int(max))) #generate a random number between 1 and what the user specified
print(y) #print it
total = total + y #update the total

print("total:", total) #tell the total to the user

Now, let's try solving a complex problem. Let's say that you wanna generate a character in Dungeons and Dragons, but you don't wanna have to go through the hassle of rolling all the dice and calculating it all up yourself. Here's a program that'll do all that for you!

import random
import math
print("This program will generate starting Dungeons and Dragons character stats using the standard 5th-edition generation method, assigning stat numbers randomly.")
name = input("Give your new character a name: ")
statlist = []
statmodlist = []
statnames = ("Strength","Dexterity","Constitution","Intellegence","Wisdom","Charisma")

for x in range(6):
rolls = [random.randint(1, 6), random.randint(1, 6), random.randint(1, 6), random.randint(1, 6)]
rolls.sort(reverse = True)
print(statnames[x], "=", rolls,", dropping", rolls[3])
stat = rolls[0] + rolls[1] + rolls[2]
statlist.append(stat)

print("------------")
printed = name+"'s Stats (and Modifiers):"
print(printed)
for x in range(6):
mod = math.floor((statlist[x] - 10)/2)

modstr = str(mod)

if mod > 0:
modstr = "+" + modstr

modstr = "("+modstr+")"

statmodlist.append(modstr)

print(" ",statnames[x], "=", statlist[x], modstr)
print("------------")
printed = name+"'s Ability Scores (without proficiencies):"
print(printed)

print(" Athletics:",statmodlist[0])
print(" Acrobatics, Sleight of Hand, Stealth:", statmodlist[1])
print(" Arcana, History, Investigation, Nature, Religion:", statmodlist[3])
print(" Animal Handling, Insight, Medicine, Perception, Survival:", statmodlist[4])
print(" Deception, Intimidation, Performance, Persuasion:", statmodlist[5])
